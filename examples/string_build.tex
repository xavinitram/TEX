// String Build — construct a sanitized filename from string parameters
//
// Demonstrates string concatenation and the sanitize_filename()
// function. Combines a prefix and suffix into a clean path-safe name.
//
// Outputs: @OUT (string)

s$prefix = "image";
s$suffix = ".png";

string name = $prefix + "_processed_" + $suffix;
@OUT = sanitize_filename(name);
