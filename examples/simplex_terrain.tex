// Procedural terrain with simplex noise
// Outputs a colored terrain image and a heightmap mask

f$frequency = 4.0;   // terrain scale (higher = more features)

vec4 _ref = @ref;
float height = simplex(u * $frequency, v * $frequency) * 0.5 + 0.5;

// Add detail with a second octave
height += simplex(u * $frequency * 2.0, v * $frequency * 2.0) * 0.25;
height = clamp(height, 0.0, 1.0);

// Color by elevation
vec3 water = vec3(0.1, 0.3, 0.7);
vec3 grass = vec3(0.2, 0.6, 0.15);
vec3 rock  = vec3(0.5, 0.45, 0.4);
vec3 snow  = vec3(0.95, 0.95, 1.0);

vec3 color = water;
if (height > 0.35) {
    color = lerp(water, grass, smoothstep(0.35, 0.45, height));
}
if (height > 0.55) {
    color = lerp(grass, rock, smoothstep(0.55, 0.7, height));
}
if (height > 0.8) {
    color = lerp(rock, snow, smoothstep(0.8, 0.9, height));
}

// Multi-output: IMAGE + MASK
@terrain = vec4(color, 1.0);
@heightmap = height;
