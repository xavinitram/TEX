// Frame Blend — weighted temporal average with adjustable window
// Averages multiple frames with falloff from center frame.
// Connect an image batch or animation to @frames.
// Useful for noise reduction, motion smoothing, and ghosting effects.

i$window = 3;      // number of frames on each side (total = 2*window+1)
f$falloff = 0.5;   // weight decay per frame (1.0 = equal, 0.5 = half per step)

vec3 sum = vec3(0.0);
float weight_sum = 0.0;

for (int i = -$window; i <= $window; i++) {
    float w = pow($falloff, abs(float(i)));
    vec3 f = @frames[ix, iy, fi + i];
    sum += f * w;
    weight_sum += w;
}

@OUT = sum / weight_sum;
