// Radial chromatic aberration — lens-style RGB fringing
//
// Shifts red outward and blue inward from the center,
// simulating the way real camera lenses separate colors.

f$strength = 0.01;   // radial offset per channel

// Direction from center
float cx = u - 0.5;
float cy = v - 0.5;

// Sample each channel at a different radial distance
float r = sample(@image, 0.5 + cx * (1.0 + $strength), 0.5 + cy * (1.0 + $strength)).r;
float g = @image.g;
float b = sample(@image, 0.5 + cx * (1.0 - $strength), 0.5 + cy * (1.0 - $strength)).b;

@OUT = vec3(r, g, b);
