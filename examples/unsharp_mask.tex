// Unsharp mask — classic image sharpening technique
// Subtracts a blurred version to enhance edges
// Uses a 3x3 Gaussian blur kernel

// 3x3 Gaussian blur
float px = 1.0 / iw;
float py = 1.0 / ih;

vec3 s00 = sample(@image, u - px, v - py);
vec3 s10 = sample(@image, u,      v - py);
vec3 s20 = sample(@image, u + px, v - py);
vec3 s01 = sample(@image, u - px, v);
vec3 s11 = @image;
vec3 s21 = sample(@image, u + px, v);
vec3 s02 = sample(@image, u - px, v + py);
vec3 s12 = sample(@image, u,      v + py);
vec3 s22 = sample(@image, u + px, v + py);

// Gaussian weights: 1/16 * [[1,2,1],[2,4,2],[1,2,1]]
vec3 blurred = (s00 + s20 + s02 + s22) * 0.0625
             + (s10 + s01 + s21 + s12) * 0.125
             + s11 * 0.25;

// Unsharp mask: original + amount * (original - blurred)
float amount = 1.5;
float threshold = 0.01;

vec3 diff = s11 - blurred;
float edge_strength = length(diff);

// Only sharpen where edge is above threshold
vec3 result = s11 + diff * amount * step(threshold, edge_strength);

@OUT = vec3(clamp(result.r, 0.0, 1.0), clamp(result.g, 0.0, 1.0), clamp(result.b, 0.0, 1.0));
