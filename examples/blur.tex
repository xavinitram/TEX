// Blur — simple box blur using fetch()
//
// Averages a (2*radius+1) x (2*radius+1) neighbourhood using fetch(),
// which samples at pixel coordinates with nearest-neighbor and clamps
// to image bounds automatically.
//
// Adjust $radius: 1 = 3x3, 2 = 5x5, 3 = 7x7.
//
// Connect @image to any RGB input.

i$radius = 2;

vec3 sum = vec3(0.0);
float count = 0.0;

for (int dy = -$radius; dy <= $radius; dy++) {
    for (int dx = -$radius; dx <= $radius; dx++) {
        sum = sum + fetch(@image, ix + dx, iy + dy).rgb;
        count = count + 1.0;
    }
}
@OUT = sum / count;
