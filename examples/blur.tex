// Box blur with adjustable radius
//
// Uses nested for loops and fetch() to average a square kernel.
// fetch() clamps to image bounds automatically.
// Adjust $radius: 1 = 3x3, 2 = 5x5, 3 = 7x7.

i$radius = 2;

vec3 sum = vec3(0.0, 0.0, 0.0);
float count = 0.0;

for (int dy = -$radius; dy <= $radius; dy++) {
    for (int dx = -$radius; dx <= $radius; dx++) {
        sum += fetch(@image, ix + dx, iy + dy);
        count += 1.0;
    }
}
@OUT = sum / count;
