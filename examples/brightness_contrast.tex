// Brightness / Contrast — filmic exposure and contrast adjustment
//
// Works in scene-linear space via inverse Reinhard tonemapping:
//   1. Remove sRGB gamma
//   2. Invert Reinhard to recover scene-linear values
//   3. Apply exposure (f-stops) and power contrast
//   4. Re-apply Reinhard and sRGB gamma
//
// This gives more natural results than naive multiply/power on gamma
// values, especially in highlights.
//
// Connect @image to any RGB input.

f$exposure = 0.0;   // exposure in f-stops (-3.0 to 3.0)
f$contrast = 1.0;   // power contrast (0.0 to 2.0)

vec3 color = @image.rgb;

// 1. Remove sRGB gamma to get display-referred linear
vec3 dispLinear = pow(color, 2.2);

// 2. Invert Reinhard tonemapping to get scene linear
vec3 safeDispLinear = min(dispLinear, 0.9999);
vec3 trueLinear = safeDispLinear / (vec3(1.0) - safeDispLinear);

// 3. Apply exposure (stops multiplier)
trueLinear = trueLinear * pow2($exposure);

// 4. Apply contrast
float pivot = 0.22;
trueLinear = pivot * pow(max(trueLinear / pivot, 0.0), $contrast);

// 5. Apply forward Reinhard tonemapping
dispLinear = trueLinear / (trueLinear + 1.0);

// 6. Re-apply sRGB gamma
color = pow(dispLinear, 1.0 / 2.2);

// Final safety clamp for valid output range
@OUT = clamp(color, 0.0, 1.0);
