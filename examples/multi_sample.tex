// Multi Sample — composite multiple inputs using binding access syntax
//
// Demonstrates the three ways to read bindings:
//   @A        — bare reference (samples at current u,v)
//   @A(u, v)  — explicit bilinear sample at custom UV coords
//   @A[ix,iy] — nearest-neighbor fetch at pixel coords
//
// Blends @A and @B using a mask with smoothstep feathering.
// Connect different images to @A, @B, and a mask to @mask.

vec3 sample_a = @A(u, v).rgb;
vec3 sample_b = @B(u, v).rgb;
float mask_val = @mask[ix, iy].r;

// Feathered blend using the mask
@OUT = lerp(sample_a, sample_b, smoothstep(0.0, 1.0, mask_val));
