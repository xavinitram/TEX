// Voronoi Cells — Distance-based cellular pattern
//
// Uses the Worley noise primitives worley_f1 (distance to nearest cell
// centre) and worley_f2 (distance to second-nearest). The difference
// F2 - F1 isolates the cell edges, producing a clean wireframe look.
//
// Inputs:  (none — purely procedural)
// Outputs: @OUT (vec3)

f$scale     = 8.0;    // cell density (higher = more cells)
f$edge_width = 0.08;  // controls edge thickness via smoothstep range

float x = u * $scale;
float y = v * $scale;

float f1 = worley_f1(x, y);
float f2 = worley_f2(x, y);

// F2 - F1 is ~0 at cell boundaries, large inside cells
float edge = f2 - f1;

// Smooth threshold to get clean edges
float mask = 1.0 - smoothstep(0.0, $edge_width, edge);

// Colour: dark cells with bright edges
vec3 cell_color = vec3(f1 * 0.3, f1 * 0.4, f1 * 0.6);
vec3 edge_color = vec3(0.9, 0.95, 1.0);

@OUT = lerp(cell_color, edge_color, mask);
