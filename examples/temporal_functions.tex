// Temporal processing — motion-adaptive temporal smoothing
//
// Demonstrates user-defined functions, temporal frame access, and
// multi-output bindings. Computes a 5-frame weighted average and a
// motion mask, then blends: static areas get smoothed, moving areas
// stay sharp.
//
// Connect an image batch or animation to @frames.
//
// Outputs: @stabilized (vec3), m@motion_mask (float)

vec4 frame_at(int offset) {
    return @frames[ix, iy, fi + offset];
}

float temporal_diff(int offset) {
    vec3 diff = frame_at(0).rgb - frame_at(offset).rgb;
    return length(diff);
}

// 5-frame weighted temporal average
vec4 center = frame_at(0);
vec4 avg = frame_at(-2) * 0.1
         + frame_at(-1) * 0.2
         + center * 0.4
         + frame_at(1) * 0.2
         + frame_at(2) * 0.1;

// Motion detection mask
float motion = temporal_diff(-1) + temporal_diff(1);
m@motion_mask = clamp(motion * 3.0, 0.0, 1.0);

// Blend: static areas get smoothed, moving areas keep sharp
float blend = smoothstep(0.05, 0.2, motion);
@stabilized = lerp(avg, center, blend).rgb;
