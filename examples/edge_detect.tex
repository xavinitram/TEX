// Edge Detect — Sobel gradient magnitude
//
// Classic Sobel operator for edge detection. Computes horizontal and
// vertical luminance gradients with 3x3 convolution kernels, then
// combines them into a single edge magnitude via Pythagorean sum.
//
// Outputs both an edge image (@edges) and a scalar mask (m@edge_mask)
// for downstream compositing or matte work.

float gx = 0.0;
float gy = 0.0;

// Sobel X kernel: [-1 0 1; -2 0 2; -1 0 1]
gx += luma(fetch(@image, ix - 1, iy - 1)) * -1.0;
gx += luma(fetch(@image, ix + 1, iy - 1)) *  1.0;
gx += luma(fetch(@image, ix - 1, iy    )) * -2.0;
gx += luma(fetch(@image, ix + 1, iy    )) *  2.0;
gx += luma(fetch(@image, ix - 1, iy + 1)) * -1.0;
gx += luma(fetch(@image, ix + 1, iy + 1)) *  1.0;

// Sobel Y kernel: [-1 -2 -1; 0 0 0; 1 2 1]
gy += luma(fetch(@image, ix - 1, iy - 1)) * -1.0;
gy += luma(fetch(@image, ix,     iy - 1)) * -2.0;
gy += luma(fetch(@image, ix + 1, iy - 1)) * -1.0;
gy += luma(fetch(@image, ix - 1, iy + 1)) *  1.0;
gy += luma(fetch(@image, ix,     iy + 1)) *  2.0;
gy += luma(fetch(@image, ix + 1, iy + 1)) *  1.0;

float edge = clamp(sqrt(gx * gx + gy * gy), 0.0, 1.0);

// Multi-output: IMAGE + MASK
@edges = vec3(edge);
m@edge_mask = edge;
