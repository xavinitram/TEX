// Unsharp mask sharpening using a 3x3 kernel
//
// Approximates a Laplacian sharpen filter:
//   center pixel * 5 minus its four direct neighbors.
// fetch() clamps to image bounds automatically.

vec4 center = fetch(@image, ix, iy);
vec4 top    = fetch(@image, ix, iy - 1);
vec4 bottom = fetch(@image, ix, iy + 1);
vec4 left   = fetch(@image, ix - 1, iy);
vec4 right  = fetch(@image, ix + 1, iy);

// Laplacian sharpen: emphasize center, subtract neighbors
vec4 sharp = center * 5.0 - top - bottom - left - right;

// Clamp to valid color range
@OUT = clamp(sharp, 0.0, 1.0);
