// Procedural cloud pattern using FBM noise
// Input: @A (IMAGE — only used for resolution)
// Output: IMAGE
//
// Uses fbm() with 6 octaves to create organic cloud shapes.
// Multiply u/v by larger values for more detailed noise.

float freq = 6.0;
float n = fbm(u * freq, v * freq, 6);

// Remap from [-1, 1] to [0, 1]
float cloud = n * 0.5 + 0.5;

// Apply contrast curve for puffier clouds
cloud = smoothstep(0.3, 0.7, cloud);

// Sky blue to white
vec3 sky = vec3(0.4, 0.6, 0.9);
vec3 white = vec3(1.0, 1.0, 1.0);
@OUT = lerp(sky, white, cloud);
