// Frame Blend — simple 3-frame temporal average
//
// Averages the previous, current, and next frames to reduce temporal
// noise and flickering. Connect an image batch or animation to @frames.
// For a configurable window size and weighted falloff, see
// frame_blend_weighted.tex.

vec3 prev = fetch_frame(@frames, fi - 1, ix, iy);
vec3 curr = fetch_frame(@frames, fi, ix, iy);
vec3 next = fetch_frame(@frames, fi + 1, ix, iy);

@OUT = (prev + curr + next) / 3.0;
