// Swirl distortion using bicubic sampling
//
// Rotates pixels around the center with angle decreasing
// toward the edges via smoothstep falloff.

f$strength = 3.0;   // swirl intensity (try 0 to 10)

float cx = u - 0.5;
float cy = v - 0.5;
float dist = sqrt(cx * cx + cy * cy);
float angle = $strength * (1.0 - smoothstep(0.0, 0.5, dist));

float cosA = cos(angle);
float sinA = sin(angle);
float su = (cx * cosA - cy * sinA) + 0.5;
float sv = (cx * sinA + cy * cosA) + 0.5;

@OUT = sample_cubic(@image, su, sv);
