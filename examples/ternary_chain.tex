// Ternary Chain — multi-level luminance classification
//
// Educational example: chained ternary operators as a compact alternative
// to if-else ladders. Classifies each pixel's luminance into four zones
// (shadow, quarter-tone, midtone, highlight) and applies a different
// color treatment to each zone.
//
// Shadow:      cool blue push, slight lift
// Quarter:     desaturate and warm slightly
// Midtone:     boost saturation, neutral
// Highlight:   warm golden tint, soft rolloff
//
// Connect @image to any RGB input.

f$shadow_thresh = 0.15;     // boundary: shadow / quarter-tone
f$mid_thresh = 0.40;        // boundary: quarter-tone / midtone
f$hi_thresh = 0.75;         // boundary: midtone / highlight
f$intensity = 1.0;          // strength of the color treatment

vec3 col = @image;
float lum = luma(col);

// ── Zone classification via chained ternary ─────────────────────────
// This reads: if lum < shadow_thresh => 0, else if < mid_thresh => 1,
// else if < hi_thresh => 2, else => 3. Compact and branch-free.
int zone = (lum < $shadow_thresh) ? 0
         : (lum < $mid_thresh)    ? 1
         : (lum < $hi_thresh)     ? 2
         :                          3;

// ── Per-zone color treatment ────────────────────────────────────────
vec3 result = col;

if (zone == 0) {
    // Shadows: cool blue push, lift blacks slightly
    vec3 tint = vec3(0.85, 0.90, 1.15);
    float lift = 0.02;
    result = col * tint + vec3(lift);
} else if (zone == 1) {
    // Quarter-tones: desaturate, warm push
    float desat = 0.7;
    result = lerp(vec3(lum), col, desat);
    result = result * vec3(1.05, 1.02, 0.95);
} else if (zone == 2) {
    // Midtones: saturation boost, neutral balance
    float sat_boost = 1.3;
    result = lerp(vec3(lum), col, sat_boost);
} else {
    // Highlights: golden tint, soft rolloff toward white
    vec3 gold = vec3(1.08, 1.04, 0.92);
    result = col * gold;
    result = lerp(result, vec3(1.0), (lum - $hi_thresh) * 0.3);
}

// Blend between original and treated based on intensity
@OUT = lerp(col, clamp(result, 0.0, 1.0), $intensity);
