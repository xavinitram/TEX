// Mandelbrot Fractal — escape-time fractal with cosine palette
//
// Demonstrates user-defined functions for iterative computation.
// Maps each pixel to a point in the complex plane, iterates the
// Mandelbrot recurrence z = z^2 + c, and colors by escape time
// using a smooth cosine palette.
//
// Inputs:  (none — purely procedural)
// Outputs: @OUT (vec3)

float iterate(float zr, float zi, float cr, float ci, int max_iter) {
    float r = zr;
    float i = zi;
    float escape = 0.0;
    for (int n = 0; n < max_iter; n++) {
        float r2 = r * r - i * i + cr;
        float i2 = 2.0 * r * i + ci;
        r = r2;
        i = i2;
        if (r * r + i * i > 4.0) {
            escape = float(n) / float(max_iter);
            return escape;
        }
    }
    return 0.0;
}

vec3 palette(float t) {
    float r = 0.5 + 0.5 * cos(TAU * (t + 0.0));
    float g = 0.5 + 0.5 * cos(TAU * (t + 0.33));
    float b = 0.5 + 0.5 * cos(TAU * (t + 0.67));
    return vec3(r, g, b);
}
