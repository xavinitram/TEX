// Motion Detect — cross-frame difference for motion highlighting
//
// Compares the current frame to the previous frame in a batch and outputs
// a brightness map: white pixels indicate motion, black indicates static
// areas. The difference is amplified 4x for visibility.
//
// Connect a multi-frame image batch to @frames.

vec4 curr = fetch_frame(@frames, fi, ix, iy);
vec4 prev = fetch_frame(@frames, max(fi - 1, 0), ix, iy);
vec4 diff = abs(curr - prev);

// Convert to grayscale motion magnitude
float motion = luma(diff.rgb) * 4.0;  // amplify for visibility
motion = clamp(motion, 0.0, 1.0);

@OUT = vec3(motion);
